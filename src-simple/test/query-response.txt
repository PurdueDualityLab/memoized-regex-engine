# Format:
#   Whitespace is stripped, and "," is a special character
#   Empty lines are ignored
#   A # introduces a comment

# REGEX , INPUT , MATCH/MISMATCH
# -----   -----   ---------------

# R1 . R2
a       , a               , MATCH

.*a.*   , bab             , MATCH
ab      , ab              , MATCH
abcde   , xyzabcxyzabcde  , MATCH

# R1 | R2

a|b     , b               , MATCH
x(a|b)y , xay             , MATCH
x(a|b)y , xby             , MATCH
x(a|b)y , xcy             , MISMATCH
